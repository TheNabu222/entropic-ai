        let currentTest = null;
        let currentAnswers = [];
        
        function openTest(testId) {
            currentTest = tests[testId];
            currentAnswers = [];
            
            const modal = document.getElementById('testModal');
            const modalTitle = document.getElementById('modalTitle');
            const modalBody = document.getElementById('modalBody');
            
            modalTitle.textContent = currentTest.title;
            
            let html = '<div class="ascii-art">╔══════════════════════════════╗\n║  TEST PROTOCOL INITIATED     ║\n╚══════════════════════════════╝</div>';
            
            currentTest.questions.forEach((question, qIndex) => {
                html += `
                    <div class="question">
                        <div class="question-text">[Q${qIndex + 1}] ${question.text}</div>
                `;
                
                question.options.forEach((option, oIndex) => {
                    html += `
                        <div class="option" onclick="selectOption(${qIndex}, ${oIndex}, ${option.score})">
                            ${option.text}
                        </div>
                    `;
                });
                
                html += '</div>';
            });
            
            html += '<button class="submit-btn" onclick="submitTest()">SUBMIT FOR EVALUATION</button>';
            
            modalBody.innerHTML = html;
            modal.style.display = 'block';
        }
        
        function selectOption(questionIndex, optionIndex, score) {
            // Deselect all options for this question
            const question = document.querySelectorAll('.question')[questionIndex];
            const options = question.querySelectorAll('.option');
            options.forEach(opt => opt.classList.remove('selected'));
            
            // Select clicked option
            options[optionIndex].classList.add('selected');
            
            // Store answer
            currentAnswers[questionIndex] = score;
        }
        
        function submitTest() {
            if (currentAnswers.length !== currentTest.questions.length) {
                alert('> ERROR: Incomplete data set\n> Please answer all questions');
                return;
            }
            
            const totalScore = currentAnswers.reduce((sum, score) => sum + score, 0);
            const resultsHtml = currentTest.evaluate(totalScore);
            
            document.getElementById('modalBody').innerHTML = resultsHtml;
        }
        
        function closeModal() {
            document.getElementById('testModal').style.display = 'none';
            currentTest = null;
            currentAnswers = [];
        }
        
        // Click outside modal to close
        window.onclick = function(event) {
            const modal = document.getElementById('testModal');
            if (event.target === modal) {
                closeModal();
            }
        }
